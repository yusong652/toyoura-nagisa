```python
def format_tool_schemas(tools: List[Dict]) -> str:
    """Format tools in Anthropic's recommended JSON Schema format."""
    if not tools:
        return ""
    
    sections = ["Here are the functions available in JSONSchema format:", "<functions>"]
    for tool in tools:
        sections.append(f'<function>{json.dumps(tool)}</function>')
    sections.append("</functions>")
    return "\n".join(sections)
```
